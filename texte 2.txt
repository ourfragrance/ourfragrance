        function toggleLanguage() {
            document.body.classList.toggle('lang-ar');
            document.body.classList.toggle('lang-en');
            document.querySelectorAll('.en').forEach(el => el.style.display = document.body.classList.contains('lang-ar') ? 'none' : 'block');
            document.querySelectorAll('.ar').forEach(el => el.style.display = document.body.classList.contains('lang-ar') ? 'block' : 'none');
        }

        const cards = document.querySelectorAll('.product-card');
        const modal = document.getElementById('orderModal');
        const closeModal = document.querySelector('.close-modal');
        const modalTitle = document.getElementById('modalTitle');
        const modalTitleAr = document.getElementById('modalTitleAr');
        const modalImg = document.getElementById('modalImg');
        const sizeSelect = document.getElementById('size');

        cards.forEach(card => {
            card.addEventListener('click', () => {
                const key = card.dataset.perfume;
                const perfume = perfumes[key];

                modalTitle.textContent = perfume.titleEn;
                modalTitleAr.textContent = perfume.titleAr;
                modalImg.src = card.querySelector('img').src;

                sizeSelect.innerHTML = '<option value="">Choose / اختر...</option>';
                perfume.sizes.forEach(s => {
                    const opt = document.createElement('option');
                    opt.value = s;
                    opt.textContent = s;
                    sizeSelect.appendChild(opt);
                });

                modal.style.display = 'flex';
            });
        });

        closeModal.addEventListener('click', () => modal.style.display = 'none');
        window.addEventListener('click', e => { if (e.target === modal) modal.style.display = 'none'; });

        document.getElementById('orderForm').addEventListener('submit', e => {

            modal.style.display = 'none';
            e.target.reset();
        });
